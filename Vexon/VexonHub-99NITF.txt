local function findEquippedAxe(axeOnly)
    for axeName, axeId in pairs(axeIds) do
        if not axeOnly or (axeName == "Old Axe" or (axeName == "Good Axe" or axeName == "Strong Axe")) then
            local inventoryItem = LocalPlayer:FindFirstChild("Inventory")
            if inventoryItem then
                inventoryItem = LocalPlayer.Inventory:FindFirstChild(axeName)
            end
            if inventoryItem then
                return inventoryItem, axeId
            end
        end
    end
    return nil, nil
end

local function equipItem(itemId)
    if itemId then
        ReplicatedStorage:WaitForChild("RemoteEvents").EquipItemHandle:FireServer("FireAllClients", itemId)
    end
end

local function unequipItem(itemId)
    if itemId then
        ReplicatedStorage:WaitForChild("RemoteEvents").UnequipItemHandle:FireServer("FireAllClients", itemId)
    end
end

local function killAuraLoop()
    while killAuraEnabled do
        local humanoidRootPart = (LocalPlayer.Character or LocalPlayer.CharacterAdded:Wait()):FindFirstChild("HumanoidRootPart")
        if humanoidRootPart then
            local equippedAxe, axeId = findEquippedAxe(false)
            if equippedAxe and axeId then
                equipItem(equippedAxe)
                for _, entity in ipairs(Workspace.Characters:GetChildren()) do
                    if entity:IsA("Model") then
                        local entityPart = entity:FindFirstChildWhichIsA("BasePart")
                        if entityPart and (entityPart.Position - humanoidRootPart.Position).Magnitude <= auraRadius then
                            pcall(function()
                                ReplicatedStorage:WaitForChild("RemoteEvents").ToolDamageObject:InvokeServer(entity, equippedAxe, axeId, CFrame.new(entityPart.Position))
                            end)
                        end
                    end
                end
                task.wait(0.1)
            else
                task.wait(1)
            end
        else
            task.wait(0.5)
        end
    end
end

function createESPText(targetPart, labelText, labelColor)
    if not targetPart:FindFirstChild("ESPTexto") then
        local billboardGui = Instance.new("BillboardGui")
        billboardGui.Name = "ESPTexto"
        billboardGui.Adornee = targetPart
        billboardGui.Size = UDim2.new(0, 100, 0, 20)
        billboardGui.StudsOffset = Vector3.new(0, 2.5, 0)
        billboardGui.AlwaysOnTop = true
        billboardGui.MaxDistance = 300
        local textLabel = Instance.new("TextLabel")
        textLabel.Parent = billboardGui
        textLabel.Size = UDim2.new(1, 0, 1, 0)
        textLabel.BackgroundTransparency = 1
        textLabel.Text = labelText
        textLabel.TextColor3 = labelColor or Color3.fromRGB(255, 255, 0)
        textLabel.TextStrokeTransparency = 0.2
        textLabel.TextScaled = true
        textLabel.Font = Enum.Font.GothamBold
        billboardGui.Parent = targetPart
    end
end

local function enableESP(targetName, espType)
    local container, espColor
    if espType == "item" then
        container = workspace:FindFirstChild("Items")
        espColor = Color3.fromRGB(0, 255, 0)
    else
        if espType ~= "mob" then
            return
        end
        container = workspace:FindFirstChild("Characters")
        espColor = Color3.fromRGB(255, 255, 0)
    end
    if container then
        for _, child in ipairs(container:GetChildren()) do
            if child.Name == targetName then
                local basePart = child:IsA("BasePart") and child and child or child:FindFirstChildWhichIsA("BasePart")
                if basePart then
                    createESPText(basePart, child.Name, espColor)
                end
            end
        end
    end
end

local function disableESP(targetName, espType)
    local container
    if espType == "item" then
        container = workspace:FindFirstChild("Items")
    else
        if espType ~= "mob" then
            return
        end
        container = workspace:FindFirstChild("Characters")
    end
    if container then
        for _, child in ipairs(container:GetChildren()) do
            if child.Name == targetName then
                local basePart = child:IsA("BasePart") and child and child or child:FindFirstChildWhichIsA("BasePart")
                if basePart then
                    for _, gui in ipairs(basePart:GetChildren()) do
                        if gui:IsA("BillboardGui") and gui.Name == "ESPTexto" then
                            gui:Destroy()
                        end
                    end
                end
            end
        end
    end
end

local bringRemoteEvents = game:GetService("ReplicatedStorage"):WaitForChild("RemoteEvents")

local function bringItem(itemModel, targetPosition)
    local primaryPart = itemModel.PrimaryPart or (itemModel:FindFirstChildWhichIsA("BasePart") or itemModel:FindFirstChild("Handle"))
    if primaryPart then
        if not itemModel.PrimaryPart then
            pcall(function()
                itemModel.PrimaryPart = primaryPart
            end)
        end
        local offsetX = math.random(-3, 3)
        local offsetZ = math.random(-3, 3)
        local offsetY = math.random(5, 10)
        local finalPosition = targetPosition + Vector3.new(offsetX, offsetY, offsetZ)
        pcall(function()
            bringRemoteEvents.RequestStartDraggingItem:FireServer(itemModel)
            task.wait(itemDropDelay)
            itemModel:SetPrimaryPartCFrame(CFrame.new(finalPosition))
            task.wait(itemDropDelay)
            bringRemoteEvents.StopDraggingItem:FireServer(itemModel)
        end)
    end
end

local function bringItems(itemNames)
    local playerCharacter = game.Players.LocalPlayer.Character
    if playerCharacter then
        playerCharacter = game.Players.LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    end
    if not playerCharacter then
        return 0
    end
    local playerPosition = playerCharacter.Position
    local itemsContainer = itemsFolder
    local itemsToProcess = {}
    
    for _, item in ipairs(itemsContainer:GetChildren()) do
        for _, targetName in ipairs(itemNames) do
            if matchItemName(item.Name, targetName) then
                table.insert(itemsToProcess, item)
                break
            end
        end
    end
    
    if maxItemsToBring < #itemsToProcess then
        local shuffledItems = {}
        for i = 1, #itemsToProcess do
            shuffledItems[i] = itemsToProcess[i]
        end
        for i = #shuffledItems, 2, -1 do
            local j = math.random(i)
            local temp = shuffledItems[j]
            shuffledItems[j] = shuffledItems[i]
            shuffledItems[i] = temp
        end
        itemsToProcess = { table.unpack(shuffledItems, 1, maxItemsToBring) }
    end
    
    local processedCount = 0
    coroutine.wrap(function()
        local currentIndex = 1
        while currentIndex <= #itemsToProcess do
            local batchCount = 0
            local batchTasks = {}
            while batchCount < batchSize and currentIndex <= #itemsToProcess do
                local currentItem = itemsToProcess[currentIndex]
                batchTasks[#batchTasks + 1] = task.spawn(function()
                    bringItem(currentItem, playerPosition)
                end)
                currentIndex = currentIndex + 1
                batchCount = batchCount + 1
                processedCount = processedCount + 1
            end
            task.wait(batchCooldown)
        end
        WindUI:Notify({
            ["Title"] = "Items Brought",
            ["Content"] = string.format("Successfully brought %d items!", processedCount),
            ["Duration"] = 2,
            ["Icon"] = "check"
        })
    end)()
    return processedCount
end

local feedRemoteEvents = game:GetService("ReplicatedStorage"):WaitForChild("RemoteEvents")
local feedItemsFolder = workspace:WaitForChild("Items")

local function getHungerLevel()
    local playerGui = game.Players.LocalPlayer:FindFirstChild("PlayerGui")
    if not playerGui then
        return 100
    end
    local interfaceGui = playerGui:FindFirstChild("Interface")
    if not interfaceGui then
        return 100
    end
    local statBars = interfaceGui:FindFirstChild("StatBars")
    if not statBars then
        return 100
    end
    local hungerBar = statBars:FindFirstChild("HungerBar")
    if not hungerBar then
        return 100
    end
    local barFrame = hungerBar:FindFirstChild("Bar")
    return not barFrame and 100 or math.floor(barFrame.Size.X.Scale * 100)
end

local function consumeItem(foodItem)
    pcall(function()
        feedRemoteEvents.RequestConsumeItem:InvokeServer(foodItem)
        task.wait(feedDelay)
    end)
end

local function autoFeedItems(foodNames)
    if hungerThreshold < getHungerLevel() then
        WindUI:Notify({
            ["Title"] = "Auto Feed",
            ["Content"] = "Hunger level is above threshold!",
            ["Duration"] = 2,
            ["Icon"] = "info"
        })
        return 0
    end
    
    local foodsToConsume = {}
    
    for _, item in ipairs(feedItemsFolder:GetChildren()) do
        for _, targetName in ipairs(foodNames) do
            if matchFoodName(item.Name, targetName) then
                table.insert(foodsToConsume, item)
                break
            end
        end
    end
    
    if #foodsToConsume == 0 then
        WindUI:Notify({
            ["Title"] = "Auto Feed",
            ["Content"] = "No food items found!",
            ["Duration"] = 2,
            ["Icon"] = "x"
        })
        return 0
    end
    
    if maxFeedItems < #foodsToConsume then
        local shuffledFoods = {}
        for i = 1, #foodsToConsume do
            shuffledFoods[i] = foodsToConsume[i]
        end
        for i = #shuffledFoods, 2, -1 do
            local j = math.random(i)
            local temp = shuffledFoods[j]
            shuffledFoods[j] = shuffledFoods[i]
            shuffledFoods[i] = temp
        end
        foodsToConsume = { table.unpack(shuffledFoods, 1, maxFeedItems) }
    end
    
    local consumedCount = 0
    coroutine.wrap(function()
        local currentIndex = 1
        while currentIndex <= #foodsToConsume do
            local batchCount = 0
            while batchCount < feedBatchSize and currentIndex <= #foodsToConsume do
                local currentFood = foodsToConsume[currentIndex]
                task.spawn(function()
                    consumeItem(currentFood)
                end)
                currentIndex = currentIndex + 1
                batchCount = batchCount + 1
                consumedCount = consumedCount + 1
            end
            task.wait(feedBatchCooldown)
        end
        WindUI:Notify({
            ["Title"] = "Auto Feed Complete",
            ["Content"] = string.format("Consumed %d food items!", consumedCount),
            ["Duration"] = 2,
            ["Icon"] = "check"
        })
    end)()
    return consumedCount
end

local function moveItemToPosition(itemModel, targetPosition)
    if itemModel and (itemModel:IsDescendantOf(workspace) and (itemModel:IsA("BasePart") or itemModel:IsA("Model"))) then
        local basePart
        if itemModel:IsA("Model") then
            basePart = itemModel.PrimaryPart or itemModel:FindFirstChildWhichIsA("BasePart") or (itemModel:FindFirstChild("Handle") or itemModel)
        else
            basePart = itemModel
        end
        if basePart and basePart:IsA("BasePart") then
            if itemModel:IsA("Model") and not itemModel.PrimaryPart then
                pcall(function()
                    itemModel.PrimaryPart = basePart
                end)
            end
            pcall(function()
                game:GetService("ReplicatedStorage"):WaitForChild("RemoteEvents").RequestStartDraggingItem:FireServer(itemModel)
                if itemModel:IsA("Model") then
                    itemModel:SetPrimaryPartCFrame(CFrame.new(targetPosition))
                else
                    basePart.CFrame = CFrame.new(targetPosition)
                end
                game:GetService("ReplicatedStorage"):WaitForChild("RemoteEvents").StopDraggingItem:FireServer(itemModel)
            end)
        end
    else
        return
    end
end

local function calculateOverlapArea(centerX, centerZ, radius, minX, minZ, maxX, maxZ)
    local sampleCount = 10000
    local insideCount = 0
    for _ = 1, sampleCount do
        local angle = math.random() * 2 * math.pi
        local distance = math.sqrt(math.random()) * radius
        local pointX = centerX + distance * math.cos(angle)
        local pointZ = centerZ + distance * math.sin(angle)
        if minX <= pointX then
            if pointX <= maxX then
                if minZ <= pointZ then
                    if pointZ <= maxZ then
                        insideCount = insideCount + 1
                    end
                end
            end
        end
    end
    return insideCount / sampleCount * math.pi * radius * radius
end

local scanBodyVelocity = Instance.new("BodyVelocity")

scanBodyVelocity.Velocity = Vector3.zero

local function attachBodyVelocity()
    if scanBodyVelocity.Parent ~= scanHumanoidRootPart then
        scanBodyVelocity.Parent = scanHumanoidRootPart
    end
end

local function detachBodyVelocity()
    if scanBodyVelocity.Parent then
        scanBodyVelocity.Parent = nil
    end
end

local function moveToPosition(targetPosition)
    while scanEnabled and (scanHumanoidRootPart.Position - targetPosition).Magnitude > 5 do
        local direction = targetPosition - scanHumanoidRootPart.Position
        if direction.Magnitude == 0 then
            break
        end
        scanBodyVelocity.Velocity = direction.Unit * scanSpeed
        task.wait()
    end
end

local function runScanLoop()
    if not scanRunning then
        scanRunning = true
        explorationLabel.Visible = true
        progressBarFrame.Visible = true
        attachBodyVelocity()
        while scanEnabled and scanRadius < scanMaxRadius do
            local posX = scanCenterPosition.X + math.cos(scanAngle) * scanRadius
            local posZ = scanCenterPosition.Z + math.sin(scanAngle) * scanRadius
            moveToPosition((Vector3.new(posX, scanCenterPosition.Y, posZ)))
            scanAngle = scanAngle + scanAngleStep
            scanRadius = scanRadius + scanRadiusStep * (scanAngleStep / (2 * math.pi))
            local overlapArea = calculateOverlapArea(scanCenterPosition.X, scanCenterPosition.Z, scanRadius, mapMinX, mapMinZ, mapMaxX, mapMaxZ)
            local percentage = math.floor(overlapArea / mapArea * 100)
            local clampedPercentage = percentage > 100 and 100 or percentage
            explorationLabel.Text = "Exploration: " .. clampedPercentage .. "%"
            progressBar.Size = UDim2.new(clampedPercentage / 100, 0, 1, 0)
            task.wait()
        end
        scanBodyVelocity.Velocity = Vector3.zero
        detachBodyVelocity()

local plantingRadius = 5
local plantingAngleIncrement = math.pi / 4
local plantingHeightOffset = 0
local previewEnabled = false
local placementMethod = "Circle"

local campfireCenterPosition = (function()
    local mapFolder = workspace:FindFirstChild("Map")
    local campground = mapFolder and mapFolder:FindFirstChild("Campground")
    if campground then
        local mainFire = campground:FindFirstChild("MainFire")
        if mainFire and mainFire:IsA("Model") then
            if mainFire.PrimaryPart then
                return mainFire.PrimaryPart.Position
            else
                return mainFire:GetModelCFrame().Position
            end
        end
        for _, child in ipairs(campground:GetChildren()) do
            if child:IsA("BasePart") then
                return child.Position
            end
        end
    end
    return Vector3.new(-41.2053, 1.0633, 29.2236)
end)()

local function generatePlantingPositions(count)
    local positions = {}
    if placementMethod ~= "Circle" then
        if placementMethod == "Square" then
            local halfRadius = plantingRadius / 2
            local gridSize = 4
            local startX = campfireCenterPosition.X - 1.5 * halfRadius
            local startZ = campfireCenterPosition.Z - 1.5 * halfRadius
            for row = 1, gridSize do
                for col = 1, gridSize do
                    if row == 1 or (row == gridSize or (col == 1 or col == gridSize)) then
                        if count <= #positions then
                            break
                        end
                        local position = Vector3.new(startX + (row - 1) * halfRadius, plantingHeightOffset + 0.5, startZ + (col - 1) * halfRadius)
                        table.insert(positions, position)
                    end
                end
            end
        end
    else
        local currentRadius = plantingRadius
        local positionCount = 0
        while positionCount < count do
            local positionsPerRing = math.floor(2 * math.pi / plantingAngleIncrement)
            if positionsPerRing == 0 then
                break
            end
            for i = 0, positionsPerRing - 1 do
                if count <= positionCount then
                    break
                end
                local angle = i * plantingAngleIncrement
                local offsetX = math.cos(angle) * currentRadius
                local offsetZ = math.sin(angle) * currentRadius
                local position = campfireCenterPosition + Vector3.new(offsetX, plantingHeightOffset + 0.5, offsetZ)
                table.insert(positions, position)
                positionCount = positionCount + 1
            end
            currentRadius = currentRadius + plantingRadius
        end
    end
    return positions
end

function ui.Creator.Request(requestData)
    local HttpService = game:GetService("HttpService")
    local success, result = pcall(function()
        if not HttpService.RequestAsync then
            return {
                ["Body"] = HttpService:GetAsync(requestData.Url),
                ["StatusCode"] = 200,
                ["Success"] = true
            }
        end
        local response = HttpService:RequestAsync({
            ["Url"] = requestData.Url,
            ["Method"] = requestData.Method or "GET",
            ["Headers"] = requestData.Headers or {}
        })
        return {
            ["Body"] = response.Body,
            ["StatusCode"] = response.StatusCode,
            ["Success"] = response.Success
        }
    end)
    if success then
        return result
    end
    error("HTTP Request failed: " .. tostring(result))
end
